-- 控制变量
local isRunning = false
local currentCount = 0
local targetCount = 0
local intervalTime = 0.5
local loopCoroutine = nil

-- 更新进度条
local function updateProgress()
    if targetCount > 0 then
        local percent = currentCount / targetCount
        progressFill.Size = UDim2.new(percent, 0, 1, 0)
        
        if percent >= 1 then
            progressFill.BackgroundColor3 = Color3.fromRGB(100, 200, 100)
        elseif percent >= 0.7 then
            progressFill.BackgroundColor3 = Color3.fromRGB(200, 200, 100)
        elseif percent >= 0.3 then
            progressFill.BackgroundColor3 = Color3.fromRGB(200, 150, 80)
        else
            progressFill.BackgroundColor3 = Color3.fromRGB(200, 100, 80)
        end
    else
        progressFill.Size = UDim2.new(0, 0, 1, 0)
    end
end

-- 停止当前协程
local function stopLoop()
    isRunning = false
    if loopCoroutine then
        coroutine.close(loopCoroutine)
        loopCoroutine = nil
    end
    statusLabel.Text = "⚫ 状态: 停止"
    statusLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
    updateProgress()
end

-- 执行一次激活
local function activateOnce()
    if activateRemote and activateRemote:IsA("RemoteEvent") then
        activateRemote:FireServer()
    elseif activateRemote and activateRemote:IsA("RemoteFunction") then
        activateRemote:InvokeServer()
    else
        warn("Activate Dumbell 不是有效的远程事件!")
        stopLoop()
        statusLabel.Text = "❌ 错误: 无效的远程事件"
        statusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        return false
    end
    return true
end

-- 开始循环
local function startLoop(totalCount, delay)
    if isRunning then
        stopLoop()
    end
    
    isRunning = true
    currentCount = 0
    targetCount = math.max(1, totalCount)
    intervalTime = math.max(0.05, delay)
    
    statusLabel.Text = string.format("🟢 运行中: 0/%d", targetCount)
    statusLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
    updateProgress()
    
    loopCoroutine = coroutine.create(function()
        while isRunning and currentCount < targetCount do
            local success = activateOnce()
            if not success then
                break
            end
            
            currentCount = currentCount + 1
            statusLabel.Text = string.format("🟢 运行中: %d/%d", currentCount, targetCount)
            updateProgress()
            
            if currentCount < targetCount then
                local startWait = tick()
                while isRunning and tick() - startWait < intervalTime do
                    task.wait()
                end
            end
        end
        
        if isRunning and currentCount >= targetCount then
            statusLabel.Text = string.format("✅ 完成! 共 %d 次", targetCount)
            statusLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
            progressFill.BackgroundColor3 = Color3.fromRGB(100, 200, 100)
        elseif not isRunning then
            statusLabel.Text = "⏸ 状态: 已暂停"
            statusLabel.TextColor3 = Color3.fromRGB(255, 200, 100)
        end
        isRunning = false
        loopCoroutine = nil
    end)
    
    coroutine.resume(loopCoroutine)
end

-- 按钮事件
startButton.MouseButton1Click:Connect(function()
    if isRunning then
        stopLoop()
    end
    
    local num = tonumber(countBox.Text)
    if not num or num <= 0 then
        statusLabel.Text = "❌ 错误: 输入有效的重复次数"
        statusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        return
    end
    
    local delay = tonumber(intervalBox.Text)
    if not delay or delay < 0 then
        statusLabel.Text = "❌ 错误: 输入有效的间隔时间"
        statusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        return
    end
    
    startLoop(math.floor(num), delay)
end)

stopButton.MouseButton1Click:Connect(function()
    stopLoop()
end)

closeButton.MouseButton1Click:Connect(function()
    screenGui:Destroy()
end)

        end
    else
        progressFill.Size = UDim2.new(0, 0, 1, 0)
    end
end

-- 停止当前协程
local function stopLoop()
    isRunning = false
    if loopCoroutine then
        coroutine.close(loopCoroutine)
        loopCoroutine = nil
    end
    statusLabel.Text = "⚫ 状态: 停止"
    statusLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
    updateProgress()
end

-- 执行一次激活
local function activateOnce()
    if activateRemote and activateRemote:IsA("RemoteEvent") then
        activateRemote:FireServer()
    elseif activateRemote and activateRemote:IsA("RemoteFunction") then
        activateRemote:InvokeServer()
    else
        warn("Activate Dumbell 不是有效的远程事件!")
        stopLoop()
        statusLabel.Text = "❌ 错误: 无效的远程事件"
        statusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        return false
    end
    return true
end

-- 开始循环
local function startLoop(totalCount, delay)
    if isRunning then
        stopLoop()
    end
    
    isRunning = true
    currentCount = 0
    targetCount = math.max(1, totalCount)
    intervalTime = math.max(0.05, delay)
    
    statusLabel.Text = string.format("🟢 运行中: 0/%d", targetCount)
    statusLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
    updateProgress()
    
    loopCoroutine = coroutine.create(function()
        while isRunning and currentCount < targetCount do
            local success = activateOnce()
            if not success then
                break
            end
            
            currentCount = currentCount + 1
            statusLabel.Text = string.format("🟢 运行中: %d/%d", currentCount, targetCount)
            updateProgress()
            
            if currentCount < targetCount then
                local startWait = tick()
                while isRunning and tick() - startWait < intervalTime do
                    task.wait()
                end
            end
        end
        
        if isRunning and currentCount >= targetCount then
            statusLabel.Text = string.format("✅ 完成! 共 %d 次", targetCount)
            statusLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
            progressFill.BackgroundColor3 = Color3.fromRGB(100, 200, 100)
        elseif not isRunning then
            statusLabel.Text = "⏸ 状态: 已暂停"
            statusLabel.TextColor3 = Color3.fromRGB(255, 200, 100)
        end
        isRunning = false
        loopCoroutine = nil
    end)
    
    coroutine.resume(loopCoroutine)
end

-- 按钮事件
startButton.MouseButton1Click:Connect(function()
    if isRunning then
        stopLoop()
    end
    
    local num = tonumber(countBox.Text)
    if not num or num <= 0 then
        statusLabel.Text = "❌ 错误: 输入有效的重复次数"
        statusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        return
    end
    
    local delay = tonumber(intervalBox.Text)
    if not delay or delay < 0 then
        statusLabel.Text = "❌ 错误: 输入有效的间隔时间"
        statusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        return
    end
    
    startLoop(math.floor(num), delay)
end)

stopButton.MouseButton1Click:Connect(function()
    stopLoop()
end)

closeButton.MouseButton1Click:Connect(function()
    screenGui:Destroy()
end)

-- 初始化
stopLoop()
